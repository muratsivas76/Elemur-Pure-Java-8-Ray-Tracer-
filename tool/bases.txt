public interface PBRCapableMaterial {
    Color getAlbedo();
    double getRoughness();
    double getMetalness();
    MaterialType getMaterialType(); // METAL/DIELECTRIC
    double getReflectivity();
    double getIndexOfRefraction();
    double getTransparency();
}
